dependency_crosswalk:

    role:
      - mainbone

      - canonical only after source_ref and path role are clear

      - keep draft in reserve or local preview before promotion

  drive_source_reservoir:
    role:
      - source reservoir
      - mirror layer
      - stemcell nuclei carrier

    sovereignty: primary_for_raw_source_not_runtime_truth

      - source exists is not enough

      - freeze in StemCell_Vault or keep as source-only node

  web_projection_shell:
    role:
      - visible portal
